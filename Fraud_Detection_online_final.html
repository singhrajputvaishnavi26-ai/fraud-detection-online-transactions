import matplotlib.pyplot as plt
import seaborn as sns

# Select numerical columns only
numerical_cols = df.select_dtypes(include=['int64', 'float64']).columns.tolist()

# Remove target column
if 'is_fraud' in numerical_cols:
    numerical_cols.remove('is_fraud')

print("=" * 70)
print("OUTLIER ANALYSIS")
print("=" * 70)

# Summary using IQR
outlier_summary = []

for col in numerical_cols:
    Q1 = df[col].quantile(0.25)
    Q3 = df[col].quantile(0.75)
    IQR = Q3 - Q1

    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR

    outliers = df[
        (df[col] < lower_bound) |
        (df[col] > upper_bound)
    ].shape[0]

    outlier_summary.append({
        'Feature': col,
        'Outlier_Count': outliers,
        'Outlier_Percentage': round(
            (outliers / len(df)) * 100, 2
        )
    })

outlier_df = pd.DataFrame(outlier_summary)
outlier_df = outlier_df.sort_values(
    by='Outlier_Percentage',
    ascending=False
)

print(outlier_df)

# ------------------------------------------------------------
# Boxplots for Important Numerical Features

plt.figure(figsize=(18, 12))

for i, col in enumerate(important_features, 1):
    plt.subplot(2, 3, i)
    sns.boxplot(x=df[col])
    plt.title(f'Boxplot of {col}')

plt.tight_layout()
plt.show()

##FEATURE ENGINEERING

df['high_amount'] = (df['amount'] > 20000).astype(int)
df['high_ip_risk'] = (df['ip_address_risk_score'] > 0.7).astype(int)
df['high_txn_velocity'] = (df['txn_count_last_24h'] > 5).astype(int)

df['amount_deviation'] = df['amount'] - df['avg_amount_last_24h']
df['customer_risk'] = df['past_fraud_count_customer'] + df['past_disputes_customer']
df['night_transaction'] = df['hour_of_day'].apply(lambda x: 1 if x < 6 else 0)

df['risk_score'] = (
    df['high_amount'] +
    df['high_ip_risk'] +
    df['high_txn_velocity'] +
    df['location_change_flag'] +
    df['is_international']

def find_best_threshold(y_true, y_proba):

    thresholds = np.arange(0.1, 0.9, 0.05)
    best_f1 = 0
    best_threshold = 0.5

    for t in thresholds:
        y_pred = (y_proba >= t).astype(int)
        f1 = f1_score(y_true, y_pred)

        if f1 > best_f1:
            best_f1 = f1
            best_threshold = t

    return best_threshold, best_f1

  2. Implement step-up authentication (OTP, biometric verification) for transactions classified as high risk.
  3. Continuously retrain the model with new transaction data to adapt to evolving fraud patterns.

Conclusion¶

The model prioritizes high recall to ensure fraudulent transactions are not missed. This aligns with industry practices where preventing fraud is more critical than minimizing false alerts
